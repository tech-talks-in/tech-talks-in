function LiveCode(props) {
    const [count, setCount] = useState(0);


    return (
        <div>
        <h1> {count} </h1>
        <button onClick={() => setCount(count + 1)}>Add</button>
        <span> </span>
        <button onClick={() => setCount(count - 1)}>Sub</button>
        </div>
    )
}
